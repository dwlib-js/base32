# Base32

## Install
`npm i --save @dwlib/base32`

## Usage
```javascript
// CJS
const Base32 = require('@dwlib/base32');
const Base32Decode = require('@dwlib/base32/decode');
const Base32DecodeBigInt = require('@dwlib/base32/decodeBigInt');
const Base32DecodeInt = require('@dwlib/base32/decodeInt');
const Base32DecodeInto = require('@dwlib/base32/decodeInto');
const Base32Encode = require('@dwlib/base32/encode');
const Base32EncodeBigInt = require('@dwlib/base32/encodeBigInt');
const Base32EncodeInt = require('@dwlib/base32/encodeInt');
const Base32IsValid = require('@dwlib/base32/isValid');
const Base32Normalize = require('@dwlib/base32/normalize');
const Base32Validate = require('@dwlib/base32/validate');
// ESM
import Base32 from '@dwlib/base32';
import Base32Decode from '@dwlib/base32/decode';
import Base32DecodeBigInt from '@dwlib/base32/decodeBigInt';
import Base32DecodeInt from '@dwlib/base32/decodeInt';
import Base32DecodeInto from '@dwlib/base32/decodeInto';
import Base32Encode from '@dwlib/base32/encode';
import Base32EncodeBigInt from '@dwlib/base32/encodeBigInt';
import Base32EncodeInt from '@dwlib/base32/encodeInt';
import Base32IsValid from '@dwlib/base32/isValid';
import Base32Normalize from '@dwlib/base32/normalize';
import Base32Validate from '@dwlib/base32/validate';
```

# Base32

## Install
`npm i --save @dwlib/base32`

## Usage
```javascript
// CJS
const base32 = require('@dwlib/base32');
// ESM
import Base32 from '@dwlib/base32';
import * as base32 from '@dwlib/base32';
// Module Exports
const {
  Base32,
  encode,
  encodeToBytes,
  decode,
  decodeToBytes,
  encodeBytes,
  encodeBytesToString,
  decodeBytes,
  decodeBytesToString,
  encodeText,
  encodeTextToBytes,
  decodeText,
  decodeBytesToText,
  encodeInt,
  decodeInt,
  encodeBigInt,
  decodeBigInt
} = base32;
```
